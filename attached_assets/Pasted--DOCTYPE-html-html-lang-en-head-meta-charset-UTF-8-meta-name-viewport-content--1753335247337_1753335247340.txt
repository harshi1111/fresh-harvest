    // Main App Component
    function App() {
      const [currentLang, setCurrentLang] = useState('en');
      const [inquiries, setInquiries] = useState([]);
      const [formData, setFormData] = useState({name: "", phone: "", email: "", message: "", product: "", orderType: "retail"});
      const [showConfirmation, setShowConfirmation] = useState(false);
      const [formErrors, setFormErrors] = useState({});
      const [isSubmitting, setIsSubmitting] = useState(false);
      const [activeNav, setActiveNav] = useState('home');
      const [cart, setCart] = useState([]);
      const [showCart, setShowCart] = useState(false);
      const [showMicrogreensModal, setShowMicrogreensModal] = useState(false);

      const t = translations[currentLang];

      // Language toggle with regional languages
      const toggleLanguage = () => {
        const languages = ['en', 'hi', 'mr', 'ta', 'pa'];
        const currentIndex = languages.indexOf(currentLang);
        const nextIndex = (currentIndex + 1) % languages.length;
        const newLang = languages[nextIndex];
        setCurrentLang(newLang);
        
        // Update language classes
        languages.forEach(lang => {
          const className = lang === 'en' ? 'english' : 
                          lang === 'hi' ? 'hindi' :
                          lang === 'mr' ? 'marathi' :
                          lang === 'ta' ? 'tamil' : 'punjabi';
          
          document.querySelectorAll(`.${className}`).forEach(el => {
            el.classList.toggle('hidden', newLang !== lang);
          });
        });
      };

      // Load data from localStorage
      useEffect(() => {
        const savedInquiries = localStorage.getItem('nvhAgriInquiries');
        if (savedInquiries) {
          setInquiries(JSON.parse(savedInquiries));
        }
        
        const savedCart = localStorage.getItem('nvhAgriCart');
        if (savedCart) {
          setCart(JSON.parse(savedCart));
        }

        // Show cookie consent
        setTimeout(() => {
          document.getElementById('cookie-consent').classList.remove('hidden');
        }, 2000);
      }, []);

      // Save to localStorage
      useEffect(() => {
        localStorage.setItem('nvhAgriInquiries', JSON.stringify(inquiries));
      }, [inquiries]);

      useEffect(() => {
        localStorage.setItem('nvhAgriCart', JSON.stringify(cart));
      }, [cart]);

      // Handle scroll for navigation
      useEffect(() => {
        const handleScroll = () => {
          const sections = ['home', 'products', 'business', 'contact'];
          const scrollY = window.scrollY;
          
          sections.forEach(section => {
            const element = document.getElementById(section);
            if (element) {
              const { offsetTop, offsetHeight } = element;
              if (scrollY >= offsetTop - 100 && scrollY < offsetTop + offsetHeight - 100) {
                setActiveNav(section);
              }
            }
          });
        };

        window.addEventListener('scroll', handleScroll);
        return () => window.removeEventListener('scroll', handleScroll);
      }, []);

      // Functions
      const scrollToSection = (sectionId) => {
        document.getElementById(sectionId)?.scrollIntoView({ behavior: 'smooth' });
      };

      const addToCart = (product, orderType = 'retail') => {
        const price = orderType === 'wholesale' ? product.wholesalePrice : 
                     orderType === 'bulk' ? product.bulkPrice : product.retailPrice;
        
        const cartItem = {
          ...product,
          selectedPrice: price,
          orderType: orderType,
          quantity: 1
        };

        const existingItem = cart.find(item => 
          item.name.en === product.name.en && item.orderType === orderType
        );
        
        if (existingItem) {
          setCart(cart.map(item => 
            item.name.en === product.name.en && item.orderType === orderType
              ? {...item, quantity: item.quantity + 1}
              : item
          ));
        } else {
          setCart([...cart, cartItem]);
        }
      };

      const removeFromCart = (productName, orderType) => {
        setCart(cart.filter(item => !(item.name.en === productName && item.orderType === orderType)));
      };

      const getTotalPrice = () => {
        return cart.reduce((total, item) => {
          const price = parseInt(item.selectedPrice.replace('₹', '').split('/')[0]);
          return total + (price * item.quantity);
        }, 0);
      };

      const handlePayment = () => {
        const options = {
          key: "rzp_test_1234567890", // Replace with actual Razorpay key
          amount: getTotalPrice() * 100,
          currency: "INR",
          name: "NVH Agri Green",
          description: "Fresh Organic Produce",
          handler: function (response) {
            alert(`Payment successful! Payment ID: ${response.razorpay_payment_id}`);
            setCart([]);
            setShowCart(false);
          },
          prefill: {
            name: formData.name,
            email: formData.email,
            contact: formData.phone
          }
        };

        const rzp = new Razorpay(options);
        rzp.open();
      };

      const validateForm = () => {
        const errors = {};
        if (!formData.name.trim()) errors.name = "Name is required";
        if (!formData.phone.trim()) errors.phone = "Phone number is required";
